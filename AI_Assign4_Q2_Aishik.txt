f1=open('sales_data.txt', 'r')
print("Printing the content of the files using read() method")
print("File contains: ")
con=f1.read()
print(con)
f1.close()
print("File has been read and closed.")

f2=open('sales_data.txt', 'r')
print("\nPrinting the content of the files using readline() method")
print("Printing the 1st line of the file: ")
con=f2.readline()
print(con)
f2.close()
print("File has been read and closed.")

f3=open('sales_data.txt', 'r')
print("\nPrinting the content of the files using readlines() method")
print("File contains: ")
l=f3.readlines()
sv = [int(i.strip()) for i in l]
f3.close()
print("Sales List: ", sv)
print("Data Type for sales list:", type(sv))
print("File has been read and closed.")
