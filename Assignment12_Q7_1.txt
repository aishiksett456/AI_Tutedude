import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

np.random.seed(42)
df = pd.DataFrame({
    'A': np.random.normal(loc=5, scale=3, size=150),
    'B': np.random.normal(loc=15, scale=4, size=150)
})
df['B'] = df['A'] * 1.5 + np.random.normal(0, 3, size=150)

sns.regplot(x='A', y='B', data=df)
plt.title('Regression Plot: B vs A')
plt.xlabel('Values of A')
plt.ylabel('Values of B')
plt.show()
